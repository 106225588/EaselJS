 32 // used to create the instruction lists used in Graphics:
 33 function Command(f, params) {
 34 	this.f = f;
 35 	this.params = params;
 36 this.prototype.exec = function(scope) { this.f.apply(scope,this.params); }
 37 }

 39 /**
 40 * Constructs a new Graphics instance.
 41 * @param instructions Optional. This is a string that will be eval'ed in the scope of this Graphics object. This provides a mechanism for generating a vector shape from a serialized string. Ex. "beginFill('#F00');drawRect(0,0,10,10);"
 42 * @class The Graphics class exposes an easy to use API for generating vector drawing instructions and drawing them to a specified context.
 43 * Note that you can use Graphics without any dependency on the Easel framework by calling draw() directly,
 44 * or it can be used with the Shape object to draw vector graphics within the context of an Easel display list.<br/><br/>
 45 * Note that all drawing methods in Graphics return the Graphics instance, so they can be chained together. For example, the following line of code would generate the instructions to draw a rectangle with a red stroke and blue fill, then render it to the specified context2D:<br/>
 46 * myGraphics.beginStroke("#F00").beginFill("#00F").drawRect(20, 20, 100, 50).draw(myContext2D);
 47 */
 48 function Graphics(instructions) {
 49 	this.initialize(instructions);
 50 this.prototype;

 52 // static public methods:

 54 	/**
 55 	* Returns a CSS compatible color string based on the specified RGB numeric color values in the format "rgba(255,255,255,1.0)", or if alpha is null then in the format "rgb(255,255,255)". For example,
 56 	* Graphics.getRGB(50,100,150,0.5) will return "rgba(50,100,150,0.5)".
 57 	* @param r The red component for the color, between 0 and 0xFF (255).
 58 	* @param g The green component for the color, between 0 and 0xFF (255).
 59 	* @param b The blue component for the color, between 0 and 0xFF (255).
 60 	* @param alpha Optional. The alpha component for the color where 0 is fully transparent and 1 is fully opaque.
 61 	* @static
 62 	*/
 63 	Graphics.getRGB = function(r, g, b, alpha) {
 64 		if (alpha == null) {
 65 			return "rgb("+r+","+g+","+b+")";
 66 		} else {
 67 			return "rgba("+r+","+g+","+b+","+alpha+")";
 68 		}
 69 	}
 70 	
 71 	/**
 72 	* Returns a CSS compatible color string based on the specified HSL numeric color values in the format "hsla(360,100,100,1.0)", or if alpha is null then in the format "hsl(360,100,100)". For example,
 73 	* Graphics.getHSL(150,100,70) will return "hsl(150,100,70)".
 74 	* @param hue The hue component for the color, between 0 and 360.
 75 	* @param saturation The saturation component for the color, between 0 and 100.
 76 	* @param lightness The lightness component for the color, between 0 and 100.
 77 	* @param alpha Optional. The alpha component for the color where 0 is fully transparent and 1 is fully opaque.
 78 	* @static
 79 	*/
 80 	Graphics.getHSL = function(hue, saturation, lightness, alpha) {
 81 		if (alpha == null) {
 82 			return "hsl("+(hue%360)+","+saturation+"%,"+lightness+"%)";
 83 		} else {
 84 			return "hsla("+(hue%360)+","+saturation+"%,"+lightness+"%,"+alpha+")";
 85 		}
 86 	}

137 // public methods:
138 	this.draw = function(ctx) {
139 		if (this._dirty) {
140 			this._updateInstructions();
141 		}
142 		var instr = this._instructions;
143 		for (var i=0, l=instr.length; i<l; i++) {
144 			instr[i].exec(ctx);
145 		}
146 	}

268 	/**
269 	* Begins a fill with the specified color. This ends the current subpath.
270 	* @param color A CSS compatible color value (ex. "#FF0000" or "rgba(255,0,0,0.5)"). Setting to null will result in no fill.
271 	*/
272 	this.beginFill = function(color) {
273 		if (this._active) { this._newPath(); }
274 		this._fillInstructions = color ? [new Command(this._setProp, ["fillStyle", color])] : null;
275 		return this;
276 	}

278 	/**
279 	* Begins a linear gradient fill defined by the line (x0,y0) to (x1,y1). This ends the current subpath. For example, the following code defines a black to white vertical gradient ranging from 20px to 120px, and draws a square to display it:<br/>
280 	* myGraphics.beginLinearGradientFill(["#000","#FFF"], [0,1], 0, 20, 0, 120).drawRect(20,20,120,120);
281 	* @param colors An array of CSS compatible color values. For example, ["#F00","#00F"] would define a gradient drawing from red to blue.
282 	* @param ratios An array of gradient positions which correspond to the colors. For example, [0.1,0.9] would draw the first color to 10% then interpolating to the second color at 90%.
283 	* @param x0 The position of the first point defining the line that defines the gradient direction and size.
284 	* @param y0 The position of the first point defining the line that defines the gradient direction and size.
285 	* @param x1 The position of the second point defining the line that defines the gradient direction and size.
286 	* @param y1 The position of the second point defining the line that defines the gradient direction and size.
287 	*/
288 	this.beginLinearGradientFill = function(colors, ratios, x0, y0, x1, y1) {
289 		if (this._active) { this._newPath(); }
290 		var o = this._ctx.createLinearGradient(x0, y0, x1, y1);
291 		for (var i=0, l=colors.length; i<l; i++) {
292 			o.addColorStop(ratios[i],colors[i]);
293 		}
294 		this._fillInstructions = [new Command(this._setProp, ["fillStyle", o])];
295 		return this;
296 	}

341 	/**
342 	* Sets the stroke style for the current subpath. Like all drawing methods, this can be chained, so you can define the stroke style and color in a single line of code like so:
343 	* myGraphics.setStrokeStyle(8,"round").beginStroke("#F00");
344 	* @param thickness The width of the stroke.
345 	* @param caps Optional. Indicates the type of caps to use at the end of lines. One of butt, round, or square. Defaults to "butt". Also accepts the values 0 (butt), 1 (round), and 2 (square) for use with the tiny API.
346 	* @param joints Optional. Specifies the type of joints that should be used where two lines meet. One of bevel, round, or miter. Defaults to "miter". Also accepts the values 0 (miter), 1 (round), and 2 (bevel) for use with the tiny API.
347 	* @param miter Optional. If joints is set to "miter", then you can specify a miter limit ratio which controls at what point a mitered joint will be clipped.
348 	*/
349 	this.setStrokeStyle = function(thickness, caps, joints, miterLimit) {
350 		if (this._active) { this._newPath(); }
351 		this._strokeStyleInstructions = [
352 			new Command(this._setProp, ["lineWidth", (thickness == null ? "1" : thickness)]),
353 			new Command(this._setProp, ["lineCap", (caps == null ? "butt" : (isNaN(caps) ? caps : Graphics.STROKE_CAPS_MAP[caps]))]),
354 			new Command(this._setProp, ["lineJoin", (joints == null ? "miter" : (isNaN(joints) ? joints : Graphics.STROKE_JOINTS_MAP[joints]))]),
355 			new Command(this._setProp, ["miterLimit", (miterLimit == null ? "10" : miterLimit)])
356 			];
357 		return this;
358 	}

458 	/**
459 	* Draws a rounded rectangle with different corner radiuses.
460 	* @param x
461 	* @param y
462 	* @param w
463 	* @param h
464 	* @param radiusTL Top left corner radius.
465 	* @param radiusTR Top right corner radius.
466 	* @param radiusBR Bottom right corner radius.
467 	* @param radiusBL Bottom left corner radius.
468 	*/
469 	this.drawRoundRectComplex = function(x, y, w, h, radiusTL, radiusTR, radiusBR, radiusBL) {
470 		this._dirty = this._active = true;
471 		this._activeInstructions.push(
472 			new Command(this._ctx.moveTo, [x+radiusTL,y]),
473 			new Command(this._ctx.lineTo, [x+w-radiusTR,y]),
474 			new Command(this._ctx.arc, [x+w-radiusTR, y+radiusTR, radiusTR, (-Math.PI/2), 0, false]),
475 			new Command(this._ctx.lineTo, [x+w, y+h-radiusBR]),
476 			new Command(this._ctx.arc, [x+w-radiusBR, y+h-radiusBR, radiusBR, 0, Math.PI/2, false]),
477 			new Command(this._ctx.lineTo, [x+radiusBL, y+h]),
478 			new Command(this._ctx.arc, [x+radiusBL, y+h-radiusBL, radiusBL, Math.PI/2, Math.PI, false]),
479 			new Command(this._ctx.lineTo, [x, y+radiusTL]),
480 			new Command(this._ctx.arc, [x+radiusTL, y+radiusTL, radiusTL, Math.PI, Math.PI*3/2, false])
481 		);
482 		return this;
483 	}

488 	* @param y

492 		this.arc(x, y, radius, 0, Math.PI*2);

496 	/**
497 	* Draws an ellipse (oval).
498 	* @param x
499 	* @param y
500 	* @param w
501 	* @param h
502 	*/
503 	this.drawEllipse = function(x, y, w, h) {
504 		this._dirty = this._active = true;
505 		var k = 0.5522848;
506 		var ox = (w / 2) * k;
507 		var oy = (h / 2) * k;
508 		var xe = x + w;
509 		var ye = y + h;
510 		var xm = x + w / 2;
511 		var ym = y + h / 2;
512 			
513 		this._activeInstructions.push(
514 			new Command(this._ctx.moveTo, [x, ym]),
515 			new Command(this._ctx.bezierCurveTo, [x, ym-oy, xm-ox, y, xm, y]),
516 			new Command(this._ctx.bezierCurveTo, [xm+ox, y, xe, ym-oy, xe, ym]),
517 			new Command(this._ctx.bezierCurveTo, [xe, ym+oy, xm+ox, ye, xm, ye]),
518 			new Command(this._ctx.bezierCurveTo, [xm-ox, ye, x, ym+oy, x, ym])
519 		);
520 		return this;
521 	}

523 	/**
524 	* Draws a star if pointSize is greater than 0 or a regular polygon if pointSize is 0 with the specified number of points.
525 	* For example, the following code will draw a familiar 5 pointed star shape centered at 100,100 and with a radius of 50:
526 	* myGraphics.beginFill("#FF0").drawPolyStar(100, 100, 50, 5, 0.6, -90); // -90 makes the first point vertical
527 	* @param x Position of the center of the shape.
528 	* @param y Position of the center of the shape.
529 	* @param radius The outer radius of the shape.
530 	* @param sides The number of points on the star or sides on the polygon.
531 	* @param pointSize The depth or "pointy-ness" of the star points. A pointSize of 0 will draw a regular polygon (no points), a pointSize of 1 will draw nothing because the points are infinitely pointy.
532 	* @param angle The angle of the first point / corner. For example a value of 0 will draw the first point directly to the right of the center.
533 	*/
534 	this.drawPolyStar = function(x, y, radius, sides, pointSize, angle) {
535 		this._dirty = this._active = true;
536 		if (pointSize == null) { pointSize = 0; }
537 		pointSize = 1-pointSize;
538 		if (angle == null) { angle = 0; }
539 		else { angle /= 180/Math.PI; }
540 		var a = Math.PI/sides;
541 		
542 		this._activeInstructions.push(new Command(this._ctx.moveTo, [x+Math.cos(angle)*radius, y+Math.sin(angle)*radius]));
543 		for (i=0; i<sides; i++) {
544 			angle += a;
545 			if (pointSize != 1) {
546 				this._activeInstructions.push(new Command(this._ctx.lineTo, [x+Math.cos(angle)*radius*pointSize, y+Math.sin(angle)*radius*pointSize]));
547 			}
548 			angle += a;
549 			this._activeInstructions.push(new Command(this._ctx.lineTo, [x+Math.cos(angle)*radius, y+Math.sin(angle)*radius]));
550 		}
551 		return this;
552 	}

628 // private methods:
629 	/** @private */
630 	this._updateInstructions = function() {
631 		this._instructions = this._oldInstructions.slice()
632 		this._instructions.push(Graphics.beginCmd);
633 		 
634 		if (this._fillInstructions) { this._instructions.push.apply(this._instructions,this._fillInstructions); }
635 		if (this._strokeInstructions) {
636 			this._instructions.push.apply(this._instructions,this._strokeInstructions);
637 			if (this._strokeStyleInstructions) {
638 				this._instructions.push.apply(this._instructions,this._strokeStyleInstructions);
639 			}
640 		}
641 		
642 		this._instructions.push.apply(this._instructions,this._activeInstructions);
643 		
644 		if (this._fillInstructions) { this._instructions.push(Graphics.fillCmd); }
645 		if (this._strokeInstructions) { this._instructions.push(Graphics.strokeCmd); }
646 	}

648 	/** @private */
649 	this._newPath = function() {
650 		if (this._dirty) { this._updateInstructions(); }
651 		this._oldInstructions = this._instructions;
652 		this._activeInstructions = [];
653 		this._active = this._dirty = false;
654 	}
