 32 /**
 33 * Constructs a Stage object with the specified target canvas.
 34 * @param canvas The canvas the stage will render to.
 35 * @class A stage is the root level Container for a display list. Each time its tick method is called, it will render its display list to its target canvas.
 36 * @augments Container
 37 */
 38 function Stage(canvas) {
 39   this.initialize(canvas);
 40 this.prototype = new Container();
 41 
 42 // static properties:
 43 	/** @private */
 44 	Stage._snapToPixelEnabled = false; // snapToPixelEnabled is temporarily copied here during a draw to provide global access.
 45 
 46 // public properties:
 47 	/** Indicates whether the stage should automatically clear the canvas before each render. You can set this to false to manually control clearing (for generative art, or when pointing multiple stages at the same canvas for example). */
 48 	this.autoClear = true;
 49 	/** The canvas the stage will render to. Multiple stages can share a single canvas, but you must disable autoClear for all but the first stage that will be ticked (or they will clear each other's render). */
 50 	this.canvas = null;
 51 	/** READ-ONLY. The current mouse X position on the canvas. If the mouse leaves the canvas, this will indicate the most recent position over the canvas. */
 52 	this.mouseX = null;
 53 	/** READ-ONLY. The current mouse Y position on the canvas. If the mouse leaves the canvas, this will indicate the most recent position over the canvas. */
 54 	this.mouseY = null;
 55 	/** The onMouseMove callback is called when the user moves the mouse over the canvas.  The handler is passed a single param containing the corresponding MouseEvent instance. */
 56 	this.onMouseMove = null;
 57 	/** The onMouseUp callback is called when the user releases the mouse button anywhere that the page can detect it.  The handler is passed a single param containing the corresponding MouseEvent instance. */
 58 	this.onMouseUp = null;
 59 	/** The onMouseDown callback is called when the user presses the mouse button over the canvas.  The handler is passed a single param containing the corresponding MouseEvent instance. */
 60 	this.onMouseDown = null;
 61 	/** Indicates whether this stage should use the snapToPixel property of display objects when rendering them. */
 62 	this.snapToPixelEnabled = false;
 63 	
 64 // private properties:
 65 	/** @private */
 66 	this._tmpCanvas = null;
 67 	/** @private */
 68 	this._activeMouseEvent = null;
 69 	/** @private */
 70 	this._activeMouseTarget = null;
 71 	/** @private */
 72 	
 73 // constructor:
 74 	/** @private */
 75 	this.Container_initialize = this.initialize;
 76 	/** @private */
 77 	this.initialize = function(canvas) {
 78 		this.Container_initialize();
 79 		this.canvas = canvas;
 80 		this.mouseChildren = true;
 81 		
 82 		var o = this;
 83 		if (window.addEventListener) {
 84 			window.addEventListener("mouseup", function(e) { o._handleMouseUp(e); }, false);
 85 		} else if (document.addEventListener) {
 86 			document.addEventListener("mouseup", function(e) { o._handleMouseUp(e); }, false);
 87 		}
 88 		canvas.addEventListener("mousemove", function(e) { o._handleMouseMove(e); }, false);
 89 		canvas.addEventListener("mousedown", function(e) { o._handleMouseDown(e); }, false);
 90 	}
 91 	
 92 // public methods:
 93 	/**
 94 	* Each time the update method is called, the stage will tick any descendants exposing a tick method (ex. BitmapSequence) and render its entire display list to the canvas.
 95 	*/
 96 	this.update = function() {
 97 		if (!this.canvas) { return; }
 98 		if (this.autoClear) { this.clear(); }
 99 		Stage._snapToPixelEnabled = this.snapToPixelEnabled;
100 		this.draw(this.canvas.getContext("2d"), false, this.getConcatenatedMatrix(DisplayObject._workingMatrix));
101 	}
102 	
103 	/**
104 	* Calls the update method. Useful for adding stage as a listener to Ticker directly.
105 	*/
106 	this.tick = this.update;
107 	
108 	/**
109 	* Clears the target canvas. Useful if autoClear is set to false.
110 	*/
111 	this.clear = function() {
112 		if (!this.canvas) { return; }
113 		var ctx = this.canvas.getContext("2d");
114 		ctx.setTransform(1,0,0,1,0,0);
115 		ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
116 	}
117 	
118 	/**
119 	* Returns a data url that contains a Base64 encoded image of the contents of the stage. The returned data url can be 
120 	* specified as the src value of an image element.
121 	* @param backgroundColor The background color to be used for the generated image. The value can be any value HTML color
122 	* value, including HEX colors, rgb and rgba. The default value is a transparent background.
123 	* @param mimeType The MIME type of the image format to be create. The default is "image/png". If an unknown MIME type
124 	* is passed in, or if the browser does not support the specified MIME type, the default value will be used.
125 	* @returns a Base64 encoded image.
126 	* @type String
127 	*/
128 	this.toDataURL = function(backgroundColor, mimeType) {
129 		if(!mimeType) {
130 			mimeType = "image/png";
131 		}
132 
133 		var ctx = this.canvas.getContext('2d');
134 		var w = this.canvas.width;
135 		var h = this.canvas.height;
136 
137 		var data;
138 
139 		if(backgroundColor) {
140 
141 			//get the current ImageData for the canvas.
142 			data = ctx.getImageData(0, 0, w, h);
143 
144 			//store the current globalCompositeOperation
145 			var compositeOperation = ctx.globalCompositeOperation;
146 
147 			//set to draw behind current content
148 			ctx.globalCompositeOperation = "destination-over";
149 
150 			//set background color
151 			ctx.fillStyle = backgroundColor;
152 
153 			//draw background on entire canvas
154 			ctx.fillRect(0,0,w,h);
155 		}
156 
157 		//get the image data from the canvas
158 		var dataURL = this.canvas.toDataURL(mimeType);
159 
160 		if(backgroundColor) {
161 			//clear the canvas
162 			ctx.clearRect (0,0,w,h);
163 
164 			//restore it with original settings
165 			ctx.putImageData(data, 0,0);
166 
167 			//reset the globalCompositeOperation to what it was
168 			ctx.globalCompositeOperation = compositeOperation;
169 		}
170 
171 		return dataURL;
172 	}
173 	
174 	this.clone = function() {
175 		var o = new Stage(null);
176 		this.cloneProps(o);
177 		return o;
178 	}
179 		
180 	this.toString = function() {
181 		return "[Stage (name="+  this.name +")]";
182 	}
183 	
184 // private methods:
185 	
186 	this._handleMouseMove = function(e) {
187 		if (!this.canvas) {
188 			this.mouseX = this.mouseY = null;
189 			return;
190 		}
191 		if(!e){ e = window.event; }
192 		this.mouseX = e.pageX-this.canvas.offsetLeft;
193 		this.mouseY = e.pageY-this.canvas.offsetTop;
194 		
195 		var evt = new MouseEvent("onMouseMove", this.mouseX, this.mouseY);
196 		if (this.onMouseMove) { this.onMouseMove(evt); }
197 		if (this._activeMouseEvent && this._activeMouseEvent.onMouseMove) { this._activeMouseEvent.onMouseMove(evt); }
198 	}
199 	
200 	this._handleMouseUp = function(e) {
201 		var evt = new MouseEvent("onMouseUp", this.mouseX, this.mouseY);
202 		if (this.onMouseUp) { this.onMouseUp(evt); }
203 		if (this._activeMouseEvent && this._activeMouseEvent.onMouseUp instanceof Function) { this._activeMouseEvent.onMouseUp(evt); }
204 		if (this._activeMouseTarget && this._activeMouseTarget.onClick &&
205 			 this._getObjectsUnderPoint(this.mouseX, this.mouseY, null, true) == this._activeMouseTarget) {
206 			this._activeMouseTarget.onClick(new MouseEvent("onClick", this.mouseX, this.mouseY));
207 		}
208 		this._activeMouseEvent = this.activeMouseTarget = null;
209 	}
210 	
211 	this._handleMouseDown = function(e) {
212 		if (this.onMouseDown) { this.onMouseDown(new MouseEvent("onMouseDown", this.mouseX, this.mouseY)); }
213 		var target = this._getObjectsUnderPoint(this.mouseX, this.mouseY, null, true);
214 		if (target) {
215 			if (target.onPress instanceof Function) {
216 				var evt = new MouseEvent("onPress", this.mouseX, this.mouseY);
217 				target.onPress(evt);
218 				if (evt.onMouseMove || evt.onMouseUp) { this._activeMouseEvent = evt; }
219 			}
220 			this._activeMouseTarget = target;
221 		}
222 	}
223 }
